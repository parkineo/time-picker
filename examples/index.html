Installation & Usage

CDN
NPM
TypeScript

CDN Usage

<!-- CSS -->
<link rel="stylesheet" href="https://unpkg.com/@parkineo/time-picker/dist/time-picker.min.css">

<!-- HTML -->
<input type="text" id="time-input" placeholder="Select time">

<!-- JavaScript -->
<script src="https://unpkg.com/@parkineo/time-picker/dist/time-picker.min.js"></script>
<script>
new TimePicker('#time-input', {
    format: '12h',
    onChange: function(time, formattedTime) {
        console.log('Selected:', formattedTime);
    }
});
</script>

NPM Installation

# Install
npm install @parkineo/time-picker

# Import CSS
import '@parkineo/time-picker/dist/time-picker.css';

# Import JavaScript
import TimePicker from '@parkineo/time-picker';

const picker = new TimePicker('#time-input', {
    format: '24h',
    step: 15
});

TypeScript Usage

import TimePicker, { TimePickerOptions } from '@parkineo/time-picker';

const options: TimePickerOptions = {
    format: '12h',
    minTime: '09:00',
    maxTime: '17:00',
    step: 30,
    onChange: (time: string, formatted: string) => {
        console.log(`Selected: ${formatted}`);
    }
};

const picker = new TimePicker('#time-input', options);

// Type-safe API calls
const currentTime: string = picker.getValue();
picker.setValue('14:30');
